function formatMoney2(v) {

    var format = "";
    var a = removeFormat(v.toString(), ',');
    a = parseInt10(a);

    var money = a.toString();

    money = reverse(money);

    for (var i = money.length - 1; i > -1; i--) {
        if ((i + 1) % 3 == 0 && money.length - 1 != i) format += ",";
        format += money.charAt(i);
    }
    
    return format;
}

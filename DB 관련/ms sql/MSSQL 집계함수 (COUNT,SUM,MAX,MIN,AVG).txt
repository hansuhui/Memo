 - 집계함수란? 
?   레코드 집합에 대해 정해진 계산을 수행하여 단일 값을 리턴해주는 함수를 말합니다.
   보통 SELECT, GROUP BY, HAVING 이하에서 사용되어집니다.

 - COUNT 사용방법
  SELECT COUNT([컬럼명]) FROM [테이블명] WHERE [조건절]

 
 - SUM 사용방법
   SELECT SUM([컬럼명]) FROM [테이블명] WHERE [조건절]

? 
 - MAX 사용방법
   SELECT MAX([컬럼명]) FROM [테이블명] WHERE [조건절]


 - MIN 사용방법
   SELECT MIN([컬럼명]) FROM [테이블명] WHERE [조건절]
 

 - AVG 사용방법 
   SELECT AVG([컬럼명]) FROM [테이블명] WHERE [조건절]
